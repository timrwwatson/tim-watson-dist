import statistics

class Lottery_Stats():
    def __init__(self, num: int, entries: list):
        self.num = num
        self.occurances = entries
        self.num_occurances = len(self.occurances)
        self.list_between = [j-i for i, j in zip(self.occurances[:-1], self.occurances[1:])]
        self.average_between = statistics.mean(self.list_between)

We can see that the numbers 4 and 7 have almost the exact same number of appearances in the lottery and roughly the same average time between appearing and the last time it appeared. The number 9 has appeared less and so has a whole week higher average.

Along with the above stats it would be nice to see what numbers have came up in the last X number of weeks. So lets create a quick and dirty way of seeing that:

def print_x_last_draws(self, x: int=20) -> None:
    """ Formatted print where the input x is the number of draws to print,
    it will then print out the draw formatted to make it apparent which numbers came out most recently
    """
    print_str = ""
    for row in range(0, x):
        dt = self.lottery_entries[row].date.strftime('%Y-%m-%d')
        normal_balls = "".join(["---" if x not in self.lottery_entries[row].balls else str(f"{x: ^3}") for x in range(1, 51)])
        special_balls = "".join(["---" if x not in self.lottery_entries[row].stars else str(f"{x: ^3}") for x in range(1, 13)])
        print_str += f"{normal_balls} {special_balls} {dt} \n"
    print(print_str)
